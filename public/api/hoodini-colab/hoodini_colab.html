45        """Start a background heartbeat to keep Colab alive without blocking UI.
46
47        This runs in a daemon thread so trait callbacks (like the Run button)
48        remain responsive. The notebook cell can finish immediately.
49
50        Example:
51            launcher = create_launcher()
52            display(launcher)
53            launcher.keep_alive()
54        """
55
56        def _heartbeat():
57            import time
58            while True:
59                time.sleep(interval_seconds)
60                # No-op heartbeat; presence of active thread helps avoid idle drop
61
62        # Avoid spawning multiple heartbeats
63        if hasattr(self, "_keep_alive_thread") and self._keep_alive_thread.is_alive():
64            return
65
66        self._keep_alive_thread: threading.Thread = threading.Thread(target=_heartbeat, daemon=True)
67        self._keep_alive_thread.start()

 71    """Create and configure a HoodiniLauncher widget with execution handler.
 72
 73    This function sets up the launcher widget and attaches the execution handler
 74    that manages installation checks and command execution.
 75
 76    Returns:
 77        HoodiniLauncher: Configured launcher widget ready to be displayed.
 78
 79    Example:
 80        >>> from hoodini_colab import create_launcher
 81        >>> launcher = create_launcher()
 82        >>> display(launcher)
 83    """
 84    from hoodini_colab.utils import (
 85        check_hoodini_installed,
 86        check_launcher_packages,
 87        install_hoodini,
 88        install_launcher_packages,
 89    )
 90
 91    def run_hoodini(change):
 92        """Run hoodini when button is clicked."""
 93        if launcher.run_requested:
 94            launcher.run_requested = False
 95            launcher.status_state = "idle"
 96
 97            # Reset logs for a fresh run
 98            launcher.logs = ""
 99
100            def append_log(text: str):
101                """Append text to the synced logs."""
102                launcher.logs = (launcher.logs or "") + text
103
104            class _LogWriter(io.StringIO):
105                def __init__(self, cb):
106                    super().__init__()
107                    self.cb = cb
108
109                def write(self, s):
110                    self.cb(s)
111                    return len(s)
112
113                def flush(self):
114                    return
115
116            log_writer = _LogWriter(append_log)
117
118            try:
119                with redirect_stdout(log_writer), redirect_stderr(log_writer):
120                    # Mount Google Drive if requested
121                    gdrive_mount_path = None
122                    if launcher.mount_gdrive:
123                        launcher.status_state = "installing_launcher"
124                        launcher.status_message = "Mounting Google Drive..."
125                        append_log(" Mounting Google Drive...\n")
126                        try:
127                            from google.colab import drive  # type: ignore[import-untyped]
128                            gdrive_mount_path = '/content/drive'
129                            drive.mount(gdrive_mount_path)
130                            append_log(" Google Drive mounted successfully at /content/drive/MyDrive\n\n")
131                            launcher.status_state = "idle"
132                            launcher.status_message = ""
133                        except ImportError:
134                            append_log("  Google Drive mount only available in Google Colab. Skipping...\n\n")
135                            gdrive_mount_path = None
136                        except Exception as e:
137                            append_log(f"  Could not mount Google Drive: {e}\n\n")
138                            gdrive_mount_path = None
139
140                    # First, check if launcher packages are installed
141                    if not check_launcher_packages():
142                        launcher.status_state = "installing_launcher"
143                        launcher.status_message = "Downloading and installing pixi"
144                        append_log(" Launcher dependencies not found. Installing...\n")
145                        if not install_launcher_packages():
146                            append_log(" Failed to install launcher dependencies.\n")
147                            launcher.status_state = "error"
148                            launcher.status_message = "Failed to install launcher dependencies"
149                            return
150                    else:
151                        append_log(" Launcher dependencies are already installed\n\n")
152
153                    # Check if hoodini is installed
154                    if not check_hoodini_installed():
155                        launcher.status_state = "installing_hoodini"
156                        launcher.status_message = "Installing environment and downloading databases"
157                        append_log(" Hoodini not found in PATH. Installing...\n")
158                        # Pass the command so we can determine which databases to download
159                        if not install_hoodini(launcher.command, launcher):
160                            append_log(" Installation failed. Please check the errors above.\n")
161                            launcher.status_state = "error"
162                            launcher.status_message = "Hoodini installation failed"
163                            return
164                        append_log(" Hoodini installation completed\n\n")
165                    else:
166                        append_log(" Hoodini is already installed\n\n")
167
168                cmd = launcher.command
169
170                # Handle input list mode - save to file if input_list is provided
171                if launcher.input_list.strip():
172                    # Create temporary file with the input list
173                    import tempfile
174                    fd, input_list_path = tempfile.mkstemp(suffix='.txt', prefix='hoodini_input_list_')
175                    try:
176                        with open(fd, 'w') as f:
177                            f.write(launcher.input_list)
178                        append_log(f" Saved input list to: {input_list_path}\n")
179                        # Replace --input <list> with --input <file_path>
180                        import re
181                        cmd = re.sub(r'--input\s+"[^"]+"', f'--input "{input_list_path}"', cmd)
182                        cmd = re.sub(r"--input\s+'[^']+'", f"--input '{input_list_path}'", cmd)
183                        cmd = re.sub(r'--input\s+\S+', f'--input "{input_list_path}"', cmd)
184                    except Exception as e:
185                        append_log(f"  Error saving input list to file: {e}\n")
186                        os.close(fd)
187                        launcher.status_state = "error"
188                        launcher.status_message = "Failed to save input list"
189                        return
190
191                append_log(f" Running: {cmd}\n")
192                append_log("=" * 60 + "\n")
193
194                # Update status to show analysis is running
195                launcher.status_state = "running"
196                launcher.status_message = "Running Hoodini analysis"
197
198                # If we installed via pixi, use pixi run
199                hoodini_env_path = (
200                    Path("/content/hoodini_env")
201                    if Path("/content/hoodini_env").exists()
202                    else Path.home() / "hoodini_env"
203                )
204                if hoodini_env_path.exists():
205                    # Change to hoodini_env directory to run pixi commands
206                    original_dir = Path.cwd()
207                    os.chdir(hoodini_env_path)
208                    cmd = cmd.replace("hoodini ", "pixi run hoodini ")
209
210                process = subprocess.Popen(
211                    cmd,
212                    shell=True,
213                    stdout=subprocess.PIPE,
214                    stderr=subprocess.STDOUT,
215                    text=True,
216                )
217                if process.stdout:
218                    for line in process.stdout:
219                        append_log(line)
220                process.wait()
221
222                if process.returncode == 0:
223                    append_log("\n" + "=" * 60 + "\n")
224                    append_log(" Hoodini analysis completed successfully!\n")
225                    append_log("=" * 60 + "\n")
226
227                    # Try to read the HTML visualization
228                    output_folder = launcher.command.split('--output ')[-1].split()[0] if '--output' in launcher.command else 'results'
229                    html_path = Path(output_folder) / 'hoodini-viz' / 'hoodini-viz.html'
230
231                    if html_path.exists():
232                        try:
233                            with open(html_path) as f:
234                                html_content = f.read()
235                            launcher.html_output = html_content
236                            append_log(f" Interactive visualization ready: {html_path}\n")
237
238                            # Auto-download HTML in Google Colab
239                            try:
240                                from google.colab import files
241                                files.download(str(html_path))
242                                append_log("  HTML download triggered automatically\n\n")
243                            except ImportError:
244                                append_log("  Auto-download only available in Google Colab\n\n")
245                            except Exception as e:
246                                append_log(f"  Could not auto-download HTML: {e}\n\n")
247                        except Exception as e:
248                            append_log(f"  Could not read HTML visualization: {e}\n")
249                    else:
250                        append_log(f"  HTML visualization not found at {html_path}\n")
251
252                    launcher.status_state = "finished"
253                    launcher.status_message = "Check output below"
254                else:
255                    append_log(f"\n Process exited with code: {process.returncode}\n")
256                    launcher.status_state = "error"
257                    launcher.status_message = f"Process failed with exit code {process.returncode}"
258
259                # Restore original directory if we changed it
260                if hoodini_env_path.exists() and "original_dir" in locals():
261                    os.chdir(original_dir)
262            except Exception as e:
263                append_log(f" Error: {e}\n")
264                launcher.status_state = "error"
265                launcher.status_message = str(e)
266
267    launcher = HoodiniLauncher()
268    launcher.observe(run_hoodini, names=["run_requested"])
269    return launcher
