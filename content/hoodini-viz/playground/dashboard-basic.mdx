```tsx
import { HoodiniDashboard } from 'hoodini-viz';
import 'hoodini-viz/dist/hoodini-viz.css';

function App() {
  const BASE_URL = 'https://raw.githubusercontent.com/pentamorfico/hoodini-viz/main/src/data';
  
  return (
    <HoodiniDashboard
      dataPaths={{
        newick: `${BASE_URL}/small/tree.nwk`,
        gffParquet: `${BASE_URL}/small/parquet/gff.parquet`,
        hoodsParquet: `${BASE_URL}/small/parquet/hoods.parquet`,
        proteinLinksParquet: `${BASE_URL}/small/parquet/protein_links.parquet`,
        nucleotideLinksParquet: `${BASE_URL}/small/parquet/nucleotide_links.parquet`,
        domainsParquet: `${BASE_URL}/small/parquet/domains.parquet`,
        proteinMetadataParquet: `${BASE_URL}/small/parquet/protein_metadata.parquet`,
        domainsMetadataParquet: `${BASE_URL}/small/parquet/domains_metadata.parquet`,
        treeMetadataParquet: `${BASE_URL}/small/parquet/tree_metadata.parquet`,
        ncRNAMetadataParquet: `${BASE_URL}/small/parquet/ncrna_metadata.parquet`,
      }}
    />
  );
}
```

## Props

| Prop | Type | Description |
|------|------|-------------|
| `dataPaths` | `DataPaths` | Object containing URLs or paths to data files |
| `initialState` | `InitialState` | Optional initial visualization state |
| `palette` | `PaletteConfig` | Optional color palette configuration |
| `proteinLinkConfig` | `ProteinLinkConfig` | Optional protein link styling |
| `nucleotideLinkConfig` | `NucleotideLinkConfig` | Optional nucleotide link styling |

## DataPaths

```ts
interface DataPaths {
  newick: string;                      // Required: Newick tree file
  gffParquet: string;                  // Required: GFF features in parquet
  hoodsParquet: string;                // Required: Neighborhood data
  proteinLinksParquet?: string;        // Optional: Protein links
  nucleotideLinksParquet?: string;     // Optional: Nucleotide links
  domainsParquet?: string;             // Optional: Domain annotations
  proteinMetadataParquet?: string;     // Optional: Protein metadata
  domainsMetadataParquet?: string;     // Optional: Domain metadata
  treeMetadataParquet?: string;        // Optional: Tree node metadata
  ncRNAMetadataParquet?: string;       // Optional: ncRNA metadata
}
```
